    public void Load()
    {
        try {
            pat=con.prepareStatement("select *from stu3");
            ResultSet rs=pat.executeQuery();
            ResultSetMetaData rss=rs.getMetaData();
            int c;
            c=rss.getColumnCount();
            ds=(DefaultTableModel)jTable1.getModel();
            ds.setRowCount(0);
            while(rs.next())
            {
                Vector v=new Vector();
                for(int a=1;a<=c;a++)
                {
                    v.add(rs.getString("id"));
                    v.add(rs.getString("name"));
                    v.add(rs.getString("age"));
                    v.add(rs.getString("course"));
                }
                ds.addRow(v);
            }
        } catch (SQLException ex) {
            Logger.getLogger(Test11.class.getName()).log(Level.SEVERE, null, ex);
        }
    }

    private void jButton2ActionPerformed(java.awt.event.ActionEvent evt) {

         try {
        String name,age,course;
        name=txtname.getText();
        age=txtage.getText();
        course=txtcourse.getText();
        
            pat=con.prepareStatement("update stu3 set name=?,age=?,course=? where id=?");
            pat.setString(1, name);
            pat.setString(2, age);
            pat.setString(3, course);
            pat.setInt(4, id);
            pat.executeUpdate();

        }
